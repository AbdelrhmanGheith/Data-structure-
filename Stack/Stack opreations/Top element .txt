int top_element() {
    if (top == 0) {  // إذا كان المكدس فارغًا
        cout << "The stack is empty.\n";  // عرض رسالة تفيد بأن المكدس فارغ
        return -1;  // إرجاع قيمة غير صحيحة
    }
    return stack[top - 1];  // إرجاع العنصر العلوي في المكدس (الموقع الأخير قبل "top")
}
