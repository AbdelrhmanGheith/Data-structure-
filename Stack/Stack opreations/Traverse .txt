void traverse() {
    if (top == 0) {  // إذا كان المكدس فارغًا
        cout << "The stack is empty.\n";  // رسالة تخبرنا بأن المكدس فارغ
        return;  // الخروج من الدالة لأننا لا نحتاج إلى التكرار إذا كان المكدس فارغًا
    }

    cout << "There are " << top << " elements in the stack:\n";
    for (int i = top - 1; i >= 0; i--) {  // التكرار من العنصر العلوي إلى العنصر السفلي
        cout << stack[i] << "\n";  // عرض كل عنصر في المكدس
    }
}
